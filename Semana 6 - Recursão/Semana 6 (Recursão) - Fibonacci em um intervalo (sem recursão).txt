#include <stdio.h>

int numFib(n1, n2){
  int i, fib1 = 0, fib2 = 1, fibaux = 0, contador = 0;
    for(i = n1+1; i<n2;){
      fibaux = fib1+fib2;
      if(fibaux>i){
        i++;
      }else if(fibaux == i && i == 1){
        contador +=2;
        i++;
      }
      else if(fibaux == i){
        contador +=1;
        i++;
        fib1 = fib2;
        fib2 = fibaux;
      }else{
        fib1 = fib2;
        fib2 = fibaux;
      }
      
      
    }
  return contador;
}

int main(){
    int n1, n2;
    
    scanf("%i", &n1);
    scanf("%i", &n2);
    
    printf("Existem %i n√∫meros de fibonacci entre %i e %i", numFib(n1, n2), n1, n2);
}
